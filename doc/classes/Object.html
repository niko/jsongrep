def parse(input)
  while line = input.readline
    JsonGrepper.new(line, O[:noprint]).grep(O[:grep]).condition(O[:condition]).print(O[:fields]).summarize(O[:summarize])
  end
rescue EOFError
  JsonGrepper.summarize
  exit
end
