# File bin/jsongrep, line 22
def parse(input)
  trap(:INT){ JsonGrepper.summarize; exit }
  while line = input.readline
    JsonGrepper.new(line).grep(O[:grep]).condition(O[:condition]).print(O[:fields]).summarize(O[:summarize])
  end
rescue EOFError
  JsonGrepper.summarize
  exit
end
